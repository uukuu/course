\chapter{定义}
	\section{特征函数}
		设 $X$ 是全集，对 $X$ 的子集 $A$，我们记 
		$$
		\chi_A(x)=\left\{ 
			\begin{aligned}
				1, &\  & x \in A, \\ 
				0, &\  & x\in X \backslash A,
			\end{aligned}
			\right.
		$$
		并称 $\chi_A$ 为 $A$ 的特征函数。
	\section{狄利克雷函数}
	$$
	D(x)=\left\{
	\begin{aligned}
		1 &,\ & x \in \mathbb{Q}, \\
		0 &,\ & x \notin \mathbb{Q}
	\end{aligned}
	\right.
	$$
